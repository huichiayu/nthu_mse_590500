from pathlib import Path
from PIL import Image
import numpy as np

base_dir = Path("grain_dataset")

images = []
labels = []

label_map = {
    "Fine":0,
    "Medium":1,
    "Coarse":2}

for class_name in label_map:

    folder = base_dir / class_name

    for fname in folder.glob("*.png"):

        img = Image.open(fname).convert("L")
        
        img = np.array(img)
        
        images.append(img)

        labels.append(label_map[class_name])

        plt.axis('off')
        plt.show()
